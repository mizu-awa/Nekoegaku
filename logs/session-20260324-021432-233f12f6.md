   255→function getEarContribution(xRatio, params, W) {
   256→  const CAT = CFG.cat;
   257→  const Z = catPath.zones.ear;
   258→  const ear1X = (CAT.earCenterX - params.earGap / W) * W;
   259→  const ear2X = (CAT.earCenterX + params.earGap / W) * W;
   260→  const x = xRatio * W;
   261→  const ear1 = params.earHeight * Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
   262→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   263→  const span = CAT.tailX - CAT.noseX;
   264→  const t = (xRatio - CAT.noseX) / span;
   265→  const earStartT = (Z.startX - CAT.noseX) / span;
   266→  const earEndT = (Z.endX - CAT.noseX) / span;
   267→  const fadeWidth = catPath.fadeWidth;
   268→  const envLeft = Math.min(1, Math.max(0, (t - earStartT) / fadeWidth));
   269→  const envRight = Math.min(1, Math.max(0, (earEndT - t) / fadeWidth));
   270→  return (ear1 + ear2) * envLeft * envRight;

   273→function getTailContribution(xRatio, params) {

   381→function calculateScore() {
   382→  if (!catPath) return 0;
   383→
   384→  const W = CFG.canvas.width;

   386→  const SC = CFG.scoring;

   392→  const lines = [
   393→    { path: catPath.topLine,    getY: (xr) => getTopY(xr, player, W, H) },
   394→    { path: catPath.bottomLine, getY: (xr) => getBottomY(xr, player, W, H) },
   395→  ];
   396→
   397→  for (const line of lines) {
   398→    const startPx = Math.floor(line.path[0].x * W);
   399→    const endPx = Math.ceil(line.path[line.path.length - 1].x * W);
   400→
   401→    for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   402→      const xr = px / W;
   403→      const playerY = line.getY(xr);
   404→      const refY = splineY(line.path, xr) * H;
   405→      totalError += (refY - playerY) ** 2;
   406→      count++;
   407→    }
   408→  }
   409→
   410→  if (count === 0) return 0;
   411→  const rmse = Math.sqrt(totalError / count);
   412→  return Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   413→}
