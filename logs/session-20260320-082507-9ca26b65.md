## 原因: エンベロープ範囲とストローク描画範囲の不一致

[settings.json](config/settings.json) で2つの異なる「終端」が定義されていて、ズレがあります：

| パラメータ | 値 | 意味 |
|---|---|---|
| `strokeBounds.feetEndX` | **0.92** | 描画ループの終了X（ピクセル736） |
| `bottomLine.feetEndT` | **0.88** | 足の波エンベロープの終端（t値） |

[game.js](js/game.js) の `getFeetContribution` で、足の波にエンベロープ（包絡線）をかけています：

```javascript
const feetEnvRight = Math.min(1, Math.max(0, (BL.feetEndT - t) / BL.feetFadeWidth));
```

`feetEndX = 0.92` を t値に変換すると **t ≈ 0.957** ですが、`feetEndT = 0.88` なので、**t > 0.88 の領域ではエンベロープが0**になり、足の波が完全に消えます。

つまり、描画ループは `feetEndX (0.92)` まで回っているのに、足のうにょうにょは `feetEndT (0.88)` 付近で消えてしまい、そこから先はベースライン（平坦な線）だけが描かれています。

## 修正方法

`feetEndT` を大きくして `feetEndX` のt値に近づければ、足の波がもっと右まで届きます。`feetEndT` を `0.88` → `0.95` 程度に変更すれば、`feetEndX` 付近までうにょうにょするようになります。修正しますか？

    76→// ============================================================
    77→// 上の線のy座標を計算
    78→// ============================================================
    79→function getTopY(xRatio, params, W, H) {
    80→  const CAT = CFG.cat;
    81→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
    82→  if (t < 0 || t > 1) return CAT.noseY * H;
    83→
    84→  // LUTベースライン（元画像から抽出した固定パス）
    85→  const baseline = interpolateBaseline(refTopBaseline, xRatio);
    86→  if (baseline !== null) {
    87→    const ears = getEarContribution(xRatio, params, W);
    88→    const tail = getTailContribution(xRatio, params, W, H);
    89→    return baseline - ears - tail;
    90→  }
    91→
    92→  // フォールバック: 数式ベースライン
    93→  const backY = CAT.backY * H;
    94→  const noseY = CAT.noseY * H;
    95→  const faceT = Math.max(0, 1 - t * CFG.topLine.faceDecayRate);
    96→  const faceY = faceT * (noseY - backY);
    97→
    98→  const ears = getEarContribution(xRatio, params, W);
    99→  const tail = getTailContribution(xRatio, params, W, H);
   100→
   101→  return backY + faceY - ears - tail;
   102→}

   104→// ============================================================
   105→// 下の線のy座標を計算
   106→// ============================================================
   107→function getBottomY(xRatio, params, W, H) {
   108→  const CAT = CFG.cat;
   109→  const BL = CFG.bottomLine;
   110→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   111→  if (t < 0 || t > 1) return CAT.noseY * H;
   112→
   113→  // LUTベースライン（元画像から抽出した固定パス）
   114→  const baseline = interpolateBaseline(refBottomBaseline, xRatio);
   115→  if (baseline !== null) {
   116→    const feet = getFeetContribution(xRatio, params, W, H);
   117→    return baseline + feet;
   118→  }
   119→
   120→  // フォールバック: 数式ベースライン
   121→  const noseY = CAT.noseY * H;
   122→  const backY = CAT.backY * H;
   123→  const bottomBaseY = CAT.bottomY * H;
   124→
   125→  const buttT = (CAT.buttX - CAT.noseX) / (CAT.tailX - CAT.noseX);
   126→
   127→  let baseY;
   128→  if (t < BL.faceEndT) {
   129→    baseY = noseY + (bottomBaseY - noseY) * (t / BL.faceEndT);
   130→  } else {
   131→    baseY = bottomBaseY;
   132→  }
   133→
   134→  if (t > BL.buttRiseStart && t <= buttT) {
   135→    const bt = (t - BL.buttRiseStart) / (buttT - BL.buttRiseStart);
   136→    const smooth = bt * bt * (3 - 2 * bt);
   137→    baseY = bottomBaseY + (backY - bottomBaseY) * smooth;
   138→  } else if (t > buttT) {
   139→    // しっぽ領域: お尻（backY）からしっぽ先端に向かって上昇
   140→    const tailProgress = (t - buttT) / (1 - buttT);
   141→    const smooth = tailProgress * tailProgress * (3 - 2 * tailProgress);
   142→    const tailTipY = backY - getTailContribution(CAT.tailX, params, W, H);
   143→    baseY = backY + (tailTipY - backY) * smooth;
   144→  }
   145→
   146→  const feet = getFeetContribution(xRatio, params, W, H);
   147→  return baseY + feet;
   148→}

   153→function drawCat(ctx, params, color, lineWidth, alpha) {
   154→  const CAT = CFG.cat;

   156→  const SC = CFG.strokeColors;

   163→  ctx.lineJoin = 'round';
   164→
   165→  const SJ = CFG.strokeJunctions;
   166→
   167→  // ストローク1: 顔（earZoneStartX → faceBottomEndX を3次ベジェ曲線1本で描画）
   168→  ctx.strokeStyle = SC.face;
   169→  ctx.beginPath();
   170→  {

   172→    const y0 = SJ.earZoneStart_topY * H;

   174→    const y3 = SJ.faceBottomEnd_bottomY * H;

   180→    ctx.bezierCurveTo(cp1x, cp1y, cp2x, cp2y, x3, y3);
   181→  }
   182→  ctx.stroke();
   183→
   184→  // ストローク2: 耳（earZoneStartX～earZoneEndX の上の線を1ストロークで描画）
   185→  ctx.strokeStyle = SC.ears;
   186→  ctx.beginPath();
   187→  for (let px = Math.floor(SB.earZoneStartX * W); px <= Math.ceil(SB.earZoneEndX * W); px++) {

   191→    else if (px === Math.ceil(SB.earZoneEndX * W)) y = SJ.earZoneEnd_topY * H;
   192→    else                                            y = getTopY(xr, params, W, H);
   193→    if (px === Math.floor(SB.earZoneStartX * W)) ctx.moveTo(px, y);
   194→    else ctx.lineTo(px, y);
   195→  }
   196→  ctx.stroke();
   197→
   198→  // ストローク3: 背中
   199→  ctx.strokeStyle = SC.back;
   200→  ctx.beginPath();
   201→  {
   202→    const x0 = SB.earZoneEndX * W;
   203→    const y0 = SJ.earZoneEnd_topY * H;
   204→    const x1 = SB.tailTopStartX * W;
   205→    const y1 = SJ.tailTopStart_topY * H;
   206→    const cpx = (x0 + x1) / 2;
   207→    const cpy = Math.min(y0, y1) - 5;
   208→    ctx.moveTo(x0, y0);
   209→    ctx.quadraticCurveTo(cpx, cpy, x1, y1);
   210→  }
   211→  ctx.stroke();
   212→
   213→  // ストローク4: しっぽ（上の線 → 先端 → 下の線）
   214→  ctx.strokeStyle = SC.tail;
   215→  ctx.beginPath();
   216→  for (let px = Math.floor(SB.tailTopStartX * W); px <= Math.ceil(SB.tailEndX * W); px++) {
   217→    const xr = px / W;
   218→    let y;
   219→    if (px === Math.floor(SB.tailTopStartX * W))  y = SJ.tailTopStart_topY * H;
   220→    else if (px === Math.ceil(SB.tailEndX * W))    y = SJ.tailTipY * H;
   221→    else                                            y = getTopY(xr, params, W, H);
   222→    if (px === Math.floor(SB.tailTopStartX * W)) ctx.moveTo(px, y);
   223→    else ctx.lineTo(px, y);
   224→  }
   225→  ctx.stroke();
   226→
   227→  // ストローク5: お尻
   228→  ctx.strokeStyle = SC.butt;
   229→  ctx.beginPath();
   230→  {
   231→    // 下の線: tailEnd → feetEnd
   232→    const x0 = SB.tailEndX * W;
   233→    const y0 = SJ.tailTipY * H;
   234→    const x1 = SB.feetEndX * W;
   235→    const y1 = SJ.feetEnd_bottomY * H;
   236→    const cpx = (x0 + x1) / 2 - 5;
   237→    const cpy = Math.min(y0, y1) + 10;
   238→    ctx.moveTo(x0, y0);
   239→    ctx.quadraticCurveTo(cpx, cpy, x1, y1);
   240→  }
   241→  ctx.stroke();
   242→
   243→  // ストローク6: 足
   244→  ctx.strokeStyle = SC.feet;
   245→  ctx.beginPath();
   246→  for (let px = Math.floor(SB.feetStartX * W); px <= Math.ceil(SB.feetEndX * W); px++) {
   247→    const xr = px / W;
   248→    let y;
   249→    if (px === Math.floor(SB.feetStartX * W))     y = SJ.feetStart_bottomY * H;
   250→    else if (px === Math.ceil(SB.feetEndX * W))   y = SJ.feetEnd_bottomY * H;
   251→    else                                            y = getBottomY(xr, params, W, H);
   252→    if (px === Math.floor(SB.feetStartX * W)) ctx.moveTo(px, y);
   253→    else ctx.lineTo(px, y);
   254→  }
   255→  ctx.stroke();
   256→
   257→  // ストローク7: 足と顔をつなぐライン
   258→  ctx.strokeStyle = SC.connector;
   259→  ctx.beginPath();
   260→  {
   261→    const x0 = SB.faceBottomEndX * W;
   262→    const y0 = SJ.faceBottomEnd_bottomY * H;
   263→    const x1 = SB.feetStartX * W;
   264→    const y1 = SJ.feetStart_bottomY * H;
   265→    const cpx = (x0 + x1) / 2;
   266→    const cpy = Math.max(y0, y1) - 10;
   267→    ctx.moveTo(x0, y0);
   268→    ctx.quadraticCurveTo(cpx, cpy, x1, y1);
   269→  }
   270→  ctx.stroke();
   271→
   272→  // 目
   273→  ctx.fillStyle = color;
   274→  const eyeW = W * CFG.drawing.eyeWidthRatio;
   275→  const eyeH = H * CFG.drawing.eyeHeightRatio;
   276→  const eyeY = CAT.eyeY * H;

   278→  ctx.beginPath();
   279→  ctx.ellipse(CAT.eye1X * W, eyeY, eyeW, eyeH, 0, 0, Math.PI * 2);
   280→  ctx.fill();
   281→
   282→  ctx.beginPath();
   283→  ctx.ellipse(CAT.eye2X * W, eyeY, eyeW, eyeH, 0, 0, Math.PI * 2);
   284→  ctx.fill();
   285→
   286→  ctx.restore();
   287→}

   306→function extractContours() {
   307→  const W = CFG.canvas.width;
   308→  const H = CFG.canvas.height;
   309→  const CAT = CFG.cat;
   310→  const SC = CFG.scoring;
   311→
   312→  const offscreen = document.createElement('canvas');
   313→  offscreen.width = W;
   314→  offscreen.height = H;
   315→  const octx = offscreen.getContext('2d');
   316→  drawTargetImage(octx, 1.0);
   317→
   318→  const imageData = octx.getImageData(0, 0, W, H);
   319→  const pixels = imageData.data;
   320→
   321→  const startPx = Math.floor(CAT.noseX * W);
   322→  const endPx = Math.ceil(CAT.tailX * W);
   323→  const sampleCount = Math.floor((endPx - startPx) / SC.sampleStep) + 1;
   324→
   325→  refTopY = new Float64Array(sampleCount);
   326→  refBottomY = new Float64Array(sampleCount);
   327→
   328→  let idx = 0;
   329→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   330→    let topOuter = -1;
   331→    let topInner = -1;
   332→    let bottomOuter = -1;
   333→    let bottomInner = -1;
   334→
   335→    // 上の線：外端（上端）を見つけてから、線の内端（下端）を探す
   336→    for (let y = 0; y < H; y++) {
   337→      if (pixels[(y * W + px) * 4 + 3] > SC.alphaThreshold) {
   338→        topOuter = y;
   339→        // 線の内端（最初の透明ピクセル）を探す
   340→        for (let y2 = y + 1; y2 < H; y2++) {
   341→          if (pixels[(y2 * W + px) * 4 + 3] <= SC.alphaThreshold) {
   342→            topInner = y2 - 1;
   343→            break;
   344→          }
   345→        }
   346→        if (topInner === -1) topInner = topOuter;
   347→        break;
   348→      }
   349→    }
   350→
   351→    // 下の線：外端（下端）を見つけてから、線の内端（上端）を探す
   352→    for (let y = H - 1; y >= 0; y--) {
   353→      if (pixels[(y * W + px) * 4 + 3] > SC.alphaThreshold) {
   354→        bottomOuter = y;
   355→        // 線の内端（最初の透明ピクセル）を探す
   356→        for (let y2 = y - 1; y2 >= 0; y2--) {
   357→          if (pixels[(y2 * W + px) * 4 + 3] <= SC.alphaThreshold) {
   358→            bottomInner = y2 + 1;
   359→            break;
   360→          }
   361→        }
   362→        if (bottomInner === -1) bottomInner = bottomOuter;
   363→        break;
   364→      }
   365→    }
   366→
   367→    // 線の中央を基準にする
   368→    let topFound = topOuter !== -1 ? (topOuter + topInner) / 2 : -1;
   369→    let bottomFound = bottomOuter !== -1 ? (bottomOuter + bottomInner) / 2 : -1;
   370→
   371→    if (topFound === -1) topFound = idx > 0 ? refTopY[idx - 1] : CAT.backY * H;
   372→    if (bottomFound === -1) bottomFound = idx > 0 ? refBottomY[idx - 1] : CAT.bottomY * H;
   373→
   374→    refTopY[idx] = topFound;
   375→    refBottomY[idx] = bottomFound;
   376→    idx++;
   377→  }
   378→
   379→  smoothContour(refTopY, SC.smoothingRadius);
   380→  smoothContour(refBottomY, SC.smoothingRadius);
   381→}

   383→function smoothContour(arr, radius) {
   384→  const copy = new Float64Array(arr);
   385→  for (let i = radius; i < arr.length - radius; i++) {
   386→    let sum = 0;
   387→    for (let j = -radius; j <= radius; j++) sum += copy[i + j];
   388→    arr[i] = sum / (radius * 2 + 1);
   389→  }
   390→}

   395→function getEarContribution(xRatio, params, W) {
   396→  const CAT = CFG.cat;
   397→  const ear1X = (CAT.earCenterX - params.earGap / W) * W;
   398→  const ear2X = (CAT.earCenterX + params.earGap / W) * W;
   399→  const x = xRatio * W;
   400→  const ear1 = params.earHeight * Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
   401→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   402→  return ear1 + ear2;
   403→}

   405→function getTailContribution(xRatio, params, W, H) {
   406→  const CAT = CFG.cat;
   407→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   408→  if (t < 0 || t > 1) return 0;
   409→  const tailStart = CFG.topLine.tailStart;
   410→  const tailT = Math.max(0, (t - tailStart) / (1 - tailStart));
   411→  const smoothTail = tailT * tailT * (3 - 2 * tailT);
   412→  return params.tailHeight * smoothTail;
   413→}

   416→  return getTopY(xRatio, params, W, H) + getEarContribution(xRatio, params, W);
   417→}

   419→function getSingleEarContribution(xRatio, earCenterPx, params, W) {
   420→  const x = xRatio * W;
   421→  return params.earHeight * Math.exp(-((x - earCenterPx) ** 2) / (2 * params.earWidth ** 2));
   422→}

   424→function getFeetContribution(xRatio, params, W, H) {
   425→  const CAT = CFG.cat;
   426→  const BL = CFG.bottomLine;
   427→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   428→  if (t < 0 || t > 1) return 0;
   429→  const feetEnvLeft = Math.min(1, Math.max(0, (t - BL.feetStartT) / BL.feetFadeWidth));
   430→  const feetEnvRight = Math.min(1, Math.max(0, (BL.feetEndT - t) / BL.feetFadeWidth));
   431→  const feetEnv = feetEnvLeft * feetEnvRight;
   432→  const wave = params.feetAmp * Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);
   433→  return feetEnv * wave;
   434→}

   436→function buildBaselineLUT() {
   437→  if (!refTopY || !refBottomY) return;
   438→
   439→  const W = CFG.canvas.width;
   440→  const H = CFG.canvas.height;
   441→  const CAT = CFG.cat;
   442→  const SC = CFG.scoring;
   443→  const answer = CFG.answerParams;
   444→
   445→  const startPx = Math.floor(CAT.noseX * W);
   446→  const endPx = Math.ceil(CAT.tailX * W);
   447→  const sampleCount = Math.floor((endPx - startPx) / SC.sampleStep) + 1;
   448→
   449→  refTopBaseline = new Float64Array(sampleCount);
   450→  refBottomBaseline = new Float64Array(sampleCount);
   451→
   452→  let idx = 0;
   453→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   454→    const xr = px / W;
   455→    // 上線: baseline = refTopY + ears + tail (可変部分を足し戻す)
   456→    refTopBaseline[idx] = refTopY[idx]
   457→      + getEarContribution(xr, answer, W)
   458→      + getTailContribution(xr, answer, W, H);
   459→    // 下線: baseline = refBottomY - feet (可変部分を引き戻す)
   460→    refBottomBaseline[idx] = refBottomY[idx]
   461→      - getFeetContribution(xr, answer, W, H);
   462→    idx++;
   463→  }
   464→}

   466→function interpolateBaseline(lut, xRatio) {
   467→  if (!lut) return null;
   468→  const W = CFG.canvas.width;
   469→  const startPx = Math.floor(CFG.cat.noseX * W);
   470→  const px = xRatio * W;
   471→  const idx = (px - startPx) / CFG.scoring.sampleStep;
   472→  const i0 = Math.max(0, Math.floor(idx));
   473→  const i1 = Math.min(i0 + 1, lut.length - 1);
   474→  const frac = idx - Math.floor(idx);
   475→  return lut[i0] * (1 - frac) + lut[i1] * frac;
   476→}

   601→  const scoredStrokes = [
   602→    // 耳（上の線）: earZoneStartX ～ earZoneEndX

   608→  ];

   612→    const ePx = Math.ceil(stroke.endX * W);

   615→      // 端点はスキップ（描画で固定値を使っているため）
   616→      if (px <= sPx || px >= ePx) continue;

   620→      const refIdx = Math.round((px - startPx) / SC.sampleStep);
   621→      if (refIdx < 0 || refIdx >= stroke.ref.length) continue;
   622→      const refY = stroke.ref[refIdx];
   623→
   624→      totalError += (refY - playerY) ** 2;
   625→      count++;
   626→    }
   627→  }
