   225→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   226→  const span = CAT.tailX - CAT.noseX;

   229→  const earEndT = (SB.earZoneEndX - CAT.noseX) / span;

   236→function getTailContribution(xRatio, params, W, H) {
   237→  const CAT = CFG.cat;

   373→function calculateScore() {
   374→  if (!catPath) return 0;
   375→
   376→  const W = CFG.canvas.width;
   377→  const H = CFG.canvas.height;
   378→  const SB = CFG.strokeBounds;

   380→  let totalError = 0;

   382→
   383→  // スライダー影響がある3ストロークの範囲（端点を除いた中間ピクセルのみ）
   384→  const scoredStrokes = [
   385→    // 耳（上の線）: earZoneStartX ～ earZoneEndX
   386→    { startX: SB.earZoneStartX, endX: SB.earZoneEndX, getY: (xr) => getTopY(xr, player, W, H), refY: (xr) => splineY(catPath.topLine, xr) * H },
   387→    // しっぽ（上の線）: tailTopStartX ～ tailEndX
   388→    { startX: SB.tailTopStartX, endX: SB.tailEndX, getY: (xr) => getTopY(xr, player, W, H), refY: (xr) => splineY(catPath.topLine, xr) * H },
   389→    // 足（下の線）: feetStartX ～ feetEndX
   390→    { startX: SB.feetStartX, endX: SB.feetEndX, getY: (xr) => getBottomY(xr, player, W, H), refY: (xr) => splineY(catPath.bottomLine, xr) * H },
   391→  ];
   392→
   393→  for (const stroke of scoredStrokes) {
   394→    const sPx = Math.floor(stroke.startX * W);
   395→    const ePx = Math.ceil(stroke.endX * W);
   396→
   397→    for (let px = sPx; px <= ePx; px += SC.sampleStep) {
   398→      if (px <= sPx || px >= ePx) continue;
   399→
   400→      const xr = px / W;
   401→      const playerY = stroke.getY(xr);
   402→      const refY = stroke.refY(xr);
   403→
   404→      totalError += (refY - playerY) ** 2;
   405→      count++;
   406→    }
   407→  }
   408→
   409→  if (count === 0) return 0;
   410→  const rmse = Math.sqrt(totalError / count);
   411→  const score = Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   412→  return score;
   413→}

**現状の描画**: ピクセル単位で `lineTo` を呼んでいるので、線が非常に滑らか・均一。

**手書きっぽくする主な手法**:

1. **微小ノイズを加える（おすすめ・最軽量）**
   - 各ポイントのY座標にシンプルなノイズ（sin波の重ね合わせ等）を加えるだけ
   - 既存のピクセルループ内で加算1回増えるだけなので負荷はほぼゼロ
   - Perlin noiseなら自然な揺らぎになるが、簡易sin重ね合わせでも十分

2. **ポイント間引き + 曲線補間**
   - 全ピクセルではなく数ピクセルおきにサンプリングし、`quadraticCurveTo` で繋ぐ
   - 描画ポイントが減るのでむしろ今より軽くなる
   - 手描きの「角が少し丸い」感じが出る

3. **二重線（わずかにずらして2回描画）**
   - 手描きの「一発で決まらない」感じが出る
   - 描画コスト2倍だが元が軽い（800px程度）ので問題なし

**避けた方が良い手法**:
- セグメントごとに `stroke()` を呼んで太さを変える → draw call激増で重い
- roughjs等の外部ライブラリ → この用途にはオーバースペック

**結論**: 1か2（または両方の組み合わせ）なら、スライダー操作時のリアルタイム再描画でも全く問題ない重さです。やりますか?
