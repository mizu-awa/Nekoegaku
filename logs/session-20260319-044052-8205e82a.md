   333→function getTopY(xRatio, params, W, H) {

   335→  if (t < 0 || t > 1) return CAT.noseY * H;
   336→

   364→function getBottomY(xRatio, params, W, H) {
   365→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX); // 0→1
   366→  if (t < 0 || t > 1) return CAT.noseY * H;
   367→

   369→  const backY = CAT.backY * H;
   370→  const bottomBaseY = CAT.bottomY * H;
   371→  const tailTipY = backY - params.tailHeight; // しっぽ先端のy
   372→
   373→  // 顔の下輪郭（固定）: 鼻先(noseY) → 足の底(bottomBaseY) へ smoothstep 遷移
   374→  // 画像データ: x≈0.025→0.145 で遷移（t≈0→0.13）
   375→  const faceEndT = 0.13;
   376→  let baseY;
   377→  if (t < faceEndT) {
   378→    const ft = t / faceEndT;
   379→    const smooth = ft * ft * (3 - 2 * ft);
   380→    baseY = noseY + (bottomBaseY - noseY) * smooth;
   381→  } else {
   382→    baseY = bottomBaseY;
   383→  }
   384→
   385→  // お尻の急上昇（固定）: x≈0.925（t≈0.96）からしっぽ先端へ急カーブ

   387→  const buttRiseStart = 0.96;
   388→  if (t > buttRiseStart) {
   389→    const bt = (t - buttRiseStart) / (1 - buttRiseStart);
   390→    baseY = bottomBaseY + (tailTipY - bottomBaseY) * bt * bt;
   391→  }
   392→
   393→  // 足のsin波（顔とお尻の領域ではフェードアウト）
   394→  const feetStartT = 0.15;
   395→  const feetEndT = 0.95;
   396→  const feetEnvLeft = Math.min(1, Math.max(0, (t - feetStartT) / 0.05));
   397→  const feetEnvRight = Math.min(1, Math.max(0, (feetEndT - t) / 0.03));
   398→  const feetEnv = feetEnvLeft * feetEnvRight;
   399→  const wave = params.feetAmp * Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);
   400→
   401→  return baseY + feetEnv * wave;
   402→}
