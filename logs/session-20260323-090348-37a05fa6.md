   253→function getEarContribution(xRatio, params, W) {
   254→  const CAT = CFG.cat;
   255→  const Z = catPath.zones.ear;
   256→  const earGap = CFG.answerParams.earGap;
   257→  const ear1X = (CAT.earCenterX - earGap / W) * W;
   258→  const ear2X = (CAT.earCenterX + earGap / W) * W;
   259→  const x = xRatio * W;
   260→  const ear1 = params.earHeight * Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
   261→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   262→  const span = CAT.tailX - CAT.noseX;
   263→  const t = (xRatio - CAT.noseX) / span;
   264→  const earStartT = (Z.startX - CAT.noseX) / span;
   265→  const earEndT = (Z.endX - CAT.noseX) / span;
   266→  const fadeWidth = catPath.fadeWidth;
   267→  const envLeft = Math.min(1, Math.max(0, (t - earStartT) / fadeWidth));
   268→  const envRight = Math.min(1, Math.max(0, (earEndT - t) / fadeWidth));
   269→  return (ear1 + ear2) * envLeft * envRight;

   273→  const CAT = CFG.cat;
   274→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);

   379→function calculateScore() {
   380→  if (!catPath) return 0;
   381→
   382→  const W = CFG.canvas.width;
   383→  const H = CFG.canvas.height;
   384→  const SC = CFG.scoring;
   385→  let totalError = 0;
   386→  let count = 0;
   387→
   388→  // 上下の線それぞれ全範囲で採点
   389→  // （endpointFadeによりスライダー効果がない区間は自動的に誤差0）
   390→  const lines = [
   391→    { path: catPath.topLine,    getY: (xr) => getTopY(xr, player, W, H) },
   392→    { path: catPath.bottomLine, getY: (xr) => getBottomY(xr, player, W, H) },
   393→  ];
   394→
   395→  for (const line of lines) {
   396→    const startPx = Math.floor(line.path[0].x * W);
   397→    const endPx = Math.ceil(line.path[line.path.length - 1].x * W);
   398→
   399→    for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   400→      const xr = px / W;
   401→      const playerY = line.getY(xr);
   402→      const refY = splineY(line.path, xr) * H;
   403→      totalError += (refY - playerY) ** 2;
   404→      count++;
   405→    }
   406→  }
   407→
   408→  if (count === 0) return 0;
   409→  const rmse = Math.sqrt(totalError / count);
   410→  return Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   411→}
