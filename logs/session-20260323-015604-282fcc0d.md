    97→  return (2*t3 - 3*t2 + 1) * p0.y + (t3 - 2*t2 + t) * tanY0
    98→       + (-2*t3 + 3*t2) * p1.y + (t3 - t2) * tanY1;
    99→}

   287→function getEarContribution(xRatio, params, W) {
   288→  const CAT = CFG.cat;
   289→  const SB = CFG.strokeBounds;
   290→  const ear1X = (CAT.earCenterX - params.earGap / W) * W;
   291→  const ear2X = (CAT.earCenterX + params.earGap / W) * W;
   292→  const x = xRatio * W;
   293→  const ear1 = params.earHeight * Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
   294→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   295→  const span = CAT.tailX - CAT.noseX;
   296→  const t = (xRatio - CAT.noseX) / span;
   297→  const earStartT = (SB.earZoneStartX - CAT.noseX) / span;
   298→  const earEndT = (SB.earZoneEndX - CAT.noseX) / span;
   299→  const fadeWidth = CFG.bottomLine.fadeWidth;
   300→  const envLeft = Math.min(1, Math.max(0, (t - earStartT) / fadeWidth));
   301→  const envRight = Math.min(1, Math.max(0, (earEndT - t) / fadeWidth));
   302→  return (ear1 + ear2) * envLeft * envRight;
   303→}
   304→
   305→function getTailContribution(xRatio, params, W, H) {
   306→  const CAT = CFG.cat;
   307→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   308→  if (t < 0 || t > 1) return 0;
   309→  const tailStart = (CFG.strokeBounds.tailTopStartX - CAT.noseX) / (CAT.tailX - CAT.noseX);
   310→  const tailT = Math.max(0, (t - tailStart) / (1 - tailStart));
   311→  const smoothTail = tailT * tailT * (3 - 2 * tailT);
   312→  return params.tailHeight * smoothTail;
   313→}

   316→function getFeetContribution(xRatio, params, W, H) {
   317→  const CAT = CFG.cat;
   318→  const BL = CFG.bottomLine;
   319→  const SB = CFG.strokeBounds;
   320→  const span = CAT.tailX - CAT.noseX;
   321→  const t = (xRatio - CAT.noseX) / span;
   322→  if (t < 0 || t > 1) return 0;
   323→  const feetStartT = (SB.feetStartX - CAT.noseX) / span - BL.fadeWidth;
   324→  const feetEndT = (SB.feetEndX - CAT.noseX) / span + BL.fadeWidth;
   325→  const feetEnvLeft = Math.min(1, Math.max(0, (t - feetStartT) / BL.fadeWidth));
   326→  const feetEnvRight = Math.min(1, Math.max(0, (feetEndT - t) / BL.fadeWidth));
   327→  const feetEnv = feetEnvLeft * feetEnvRight;
   328→  const wave = params.feetAmp * Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);
   329→  return feetEnv * wave;
   330→}

   443→function calculateScore() {
   444→  if (!catPath) return 0;
   445→
   446→  const W = CFG.canvas.width;
   447→  const H = CFG.canvas.height;
   448→  const SB = CFG.strokeBounds;
   449→  const SC = CFG.scoring;
   450→  let totalError = 0;
   451→  let count = 0;
   452→
   453→  // スライダー影響がある3ストロークの範囲（端点を除いた中間ピクセルのみ）
   454→  const scoredStrokes = [
   455→    // 耳（上の線）: earZoneStartX ～ earZoneEndX
   456→    { startX: SB.earZoneStartX, endX: SB.earZoneEndX, getY: (xr) => getTopY(xr, player, W, H), refY: (xr) => splineY(catPath.topLine, xr) * H },
   457→    // しっぽ（上の線）: tailTopStartX ～ tailEndX
   458→    { startX: SB.tailTopStartX, endX: SB.tailEndX, getY: (xr) => getTopY(xr, player, W, H), refY: (xr) => splineY(catPath.topLine, xr) * H },
   459→    // 足（下の線）: feetStartX ～ feetEndX
   460→    { startX: SB.feetStartX, endX: SB.feetEndX, getY: (xr) => getBottomY(xr, player, W, H), refY: (xr) => splineY(catPath.bottomLine, xr) * H },
   461→  ];
   462→
   463→  for (const stroke of scoredStrokes) {
   464→    const sPx = Math.floor(stroke.startX * W);
   465→    const ePx = Math.ceil(stroke.endX * W);
   466→
   467→    for (let px = sPx; px <= ePx; px += SC.sampleStep) {
   468→      if (px <= sPx || px >= ePx) continue;
   469→
   470→      const xr = px / W;
   471→      const playerY = stroke.getY(xr);
   472→      const refY = stroke.refY(xr);
   473→
   474→      totalError += (refY - playerY) ** 2;
   475→      count++;
   476→    }
   477→  }
   478→
   479→  if (count === 0) return 0;
   480→  const rmse = Math.sqrt(totalError / count);
   481→  const score = Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   482→  return score;
   483→}
