#### **上の線（y_top）** - 頭～背中～しっぽ (324-348行)

```javascript
function getTopY(xRatio, params, W, H) {
  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX); // 正規化 (0→1)
  
  const backY = CAT.backY * H;
  const noseY = CAT.noseY * H;

  // 1. 顔の輪郭（左端で下に張り出す二次減衰）
  const faceT = Math.max(0, 1 - t * 6);
  const faceY = faceT * faceT * (noseY - backY);

  // 2. 耳のピーク（ガウシアン × 2つ）
  const ear1X = (CAT.earCenterX - params.earGap / W) * W;
  const ear2X = (CAT.earCenterX + params.earGap / W) * W;
  const x = xRatio * W;
  const ear1 = params.earHeight * 
    Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
  const ear2 = params.earHeight * 
    Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));

  // 3. しっぽ（右端で上に跳ねる二次関数）
  const tailStart = 0.82;
  const tailT = Math.max(0, (t - tailStart) / (1 - tailStart));
  const tail = params.tailHeight * tailT * tailT;

  return backY + faceY - ear1 - ear2 - tail;
}
```

**構成要素**:
- **背中ベースライン**: `backY` 定数
- **顔の輪郭**: 二次減衰 `(1 - t * 6)²`
- **耳**: ガウシアン分布 `A * exp(-(x - μ)² / (2σ²))` × 2つ
- **しっぽ**: 二次関数 `tailHeight * tailT²`

#### **下の線（y_bottom）** - あご～足～しっぽ付け根 (353-370行)

```javascript
function getBottomY(xRatio, params, W, H) {
  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
  
  const noseY = CAT.noseY * H;
  const backY = CAT.backY * H;
  const tailTipY = backY - params.tailHeight;
  
  // 1. 直線補間（顔先端 → しっぽ先端）
  const baseY = noseY * (1 - t) + tailTipY * t;

  // 2. エンベロープ（左右端で0になる）
  const env = Math.sin(Math.PI * t);

  // 3. 足のsin波
  const waveBase = (CAT.bottomY * H - noseY) * 0.7;
  const wave = params.feetAmp * 
    Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);

  return baseY + env * (waveBase + wave);
}
```

**構成要素**:
- **直線補間**: 左端と右端を結ぶ基本形
- **エンベロープ**: `sin(π * t)` で左右端が消える
- **足**: サイン波 `A * sin(ω * 2π * t + φ)`

function extractContours() {
  const W = 800, H = 400;
  
  // Render target image to an offscreen canvas
  const offscreen = document.createElement('canvas');
  offscreen.width = W;
  offscreen.height = H;
  const octx = offscreen.getContext('2d');
  drawTargetImage(octx, 1.0);
  
  // Get pixel data
  const imageData = octx.getImageData(0, 0, W, H);
  const pixels = imageData.data; // RGBA
  
  const startPx = Math.floor(CAT.noseX * W);  // 56
  const endPx = Math.ceil(CAT.tailX * W);      // 736
  
  refTopY = new Float64Array(Math.floor((endPx - startPx) / 2) + 1);
  refBottomY = new Float64Array(refTopY.length);
  
  let idx = 0;
  for (let px = startPx; px <= endPx; px += 2) {
    // Scan column top-down to find first opaque pixel (top contour)
    let topFound = H / 2, bottomFound = H / 2;
    
    for (let y = 0; y < H; y++) {
      const a = pixels[(y * W + px) * 4 + 3]; // alpha channel
      if (a > 128) { // threshold for "ink"
        topFound = y;
        break;
      }
    }
    
    // Scan column bottom-up to find first opaque pixel (bottom contour)
    for (let y = H - 1; y >= 0; y--) {
      const a = pixels[(y * W + px) * 4 + 3];
      if (a > 128) {
        bottomFound = y;
        break;
      }
    }
    
    refTopY[idx] = topFound;
    refBottomY[idx] = bottomFound;
    idx++;
  }
}
```
