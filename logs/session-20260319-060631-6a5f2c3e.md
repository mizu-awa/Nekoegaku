   308→function getFeetContribution(xRatio, params, W, H) {
   309→  const CAT = CFG.cat;
   310→  const BL = CFG.bottomLine;
   311→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   312→  if (t < 0 || t > 1) return 0;
   313→  const feetEnvLeft = Math.min(1, Math.max(0, (t - BL.feetStartT) / BL.feetFadeWidth));
   314→  const feetEnvRight = Math.min(1, Math.max(0, (BL.feetEndT - t) / BL.feetFadeWidth));
   315→  const feetEnv = feetEnvLeft * feetEnvRight;
   316→  const wave = params.feetAmp * Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);
   317→  return feetEnv * wave;
   318→}

   320→function buildBaselineLUT() {
   321→  if (!refTopY || !refBottomY) return;
   322→
   323→  const W = CFG.canvas.width;
   324→  const H = CFG.canvas.height;
   325→  const CAT = CFG.cat;
   326→  const SC = CFG.scoring;
   327→  const answer = CFG.answerParams;
   328→
   329→  const startPx = Math.floor(CAT.noseX * W);
   330→  const endPx = Math.ceil(CAT.tailX * W);
   331→  const sampleCount = Math.floor((endPx - startPx) / SC.sampleStep) + 1;
   332→
   333→  refTopBaseline = new Float64Array(sampleCount);
   334→  refBottomBaseline = new Float64Array(sampleCount);
   335→
   336→  let idx = 0;
   337→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   338→    const xr = px / W;

   340→    refTopBaseline[idx] = refTopY[idx]
   341→      + getEarContribution(xr, answer, W)
   342→      + getTailContribution(xr, answer, W, H);
   343→    // 下線: baseline = refBottomY - feet (可変部分を引き戻す)
   344→    refBottomBaseline[idx] = refBottomY[idx]
   345→      - getFeetContribution(xr, answer, W, H);
   346→    idx++;
   347→  }
   348→}

   581→window.autoFit = function () {
   582→  if (!refTopY || !refBottomY) { console.error('extractContours未実行'); return; }
   583→
   584→  const W = CFG.canvas.width;
   585→  const H = CFG.canvas.height;
   586→  const CAT = CFG.cat;
   587→  const SC = CFG.scoring;
   588→  const startPx = Math.floor(CAT.noseX * W);
   589→  const endPx = Math.ceil(CAT.tailX * W);
   590→
   591→  // 現在の数式ベースラインとの誤差を計算
   592→  function calcError(params) {
   593→    let err = 0;
   594→    let count = 0;
   595→    let idx = 0;
   596→    for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   597→      const xr = px / W;
   598→      const t = (xr - CAT.noseX) / (CAT.tailX - CAT.noseX);
   599→      if (t < 0 || t > 1) { idx++; continue; }
   600→
   601→      // 上線: refTopY = baseline - ears - tail を期待
   602→      const ears = getEarContribution(xr, params, W);
   603→      const tail = getTailContribution(xr, params, W, H);
   604→      const feet = getFeetContribution(xr, params, W, H);
   605→
   606→      // 上線の誤差（LUTなしの直接比較）
   607→      // getTopY(数式) = backY + faceY - ears - tail
   608→      // 理想: refTopY[idx] = getTopY(answerParams)
   609→      // ここでは数式ベースラインは使わず、refTopYとの差を最小化
   610→      const backY_val = CAT.backY * H;
   611→      const noseY_val = CAT.noseY * H;
   612→      const faceT = Math.max(0, 1 - t * CFG.topLine.faceDecayRate);
   613→      const faceY = faceT * (noseY_val - backY_val);
   614→      const mathTop = backY_val + faceY - ears - tail;
   615→
   616→      // ※自動フィッティングではLUTを使わず、直接refTopYとparamsでの曲線を比較
   617→      // refTopY = 画像の正解 → paramsでの曲線がこれに近いほど良い
   618→      err += (refTopY[idx] - mathTop) ** 2;
   619→
   620→      // 下線
   621→      const BL = CFG.bottomLine;
   622→      const buttT = (CAT.buttX - CAT.noseX) / (CAT.tailX - CAT.noseX);
   623→      let baseY;
   624→      if (t < BL.faceEndT) {
   625→        baseY = noseY_val + (CAT.bottomY * H - noseY_val) * (t / BL.faceEndT);
   626→      } else {
   627→        baseY = CAT.bottomY * H;
   628→      }
   629→      if (t > BL.buttRiseStart && t <= buttT) {
   630→        const bt = (t - BL.buttRiseStart) / (buttT - BL.buttRiseStart);
   631→        const smooth = bt * bt * (3 - 2 * bt);
   632→        baseY = CAT.bottomY * H + (backY_val - CAT.bottomY * H) * smooth;
   633→      } else if (t > buttT) {
   634→        baseY = backY_val;
   635→      }
   636→      const mathBot = baseY + feet;
   637→      err += (refBottomY[idx] - mathBot) ** 2;
   638→
   639→      count += 2;
   640→      idx++;
   641→    }
   642→    return Math.sqrt(err / count);
   643→  }
   644→
   645→  // グリッドサーチ + 局所最適化
   646→  let best = { ...CFG.answerParams };
   647→  let bestErr = calcError(best);
   648→  console.log('初期誤差:', bestErr.toFixed(2));
   649→
   650→  // 各パラメータの探索範囲

   652→    earHeight: [30, 150, 5],
   653→    earGap:    [5, 60, 3],

   656→    feetAmp:   [10, 70, 3],
   657→    feetFreq:  [2.0, 8.0, 0.2],
   658→    feetPhase: [-3.14, 3.14, 0.15]
   659→  };
   660→
   661→  // 3ラウンドの反復最適化
   662→  for (let round = 0; round < 3; round++) {
   663→    for (const [key, [min, max, step]] of Object.entries(ranges)) {
   664→      let localBest = best[key];
   665→      let localBestErr = bestErr;
   666→      const s = step / (round + 1); // ラウンドごとにステップを細かく
   667→      for (let v = min; v <= max; v += s) {
   668→        const trial = { ...best, [key]: v };
   669→        const e = calcError(trial);
   670→        if (e < localBestErr) {
   671→          localBestErr = e;
   672→          localBest = v;
   673→        }
   674→      }
   675→      best[key] = localBest;
   676→      bestErr = localBestErr;
   677→    }
   678→    console.log(`ラウンド${round + 1} 誤差: ${bestErr.toFixed(2)}`);
   679→  }
   680→
   681→  // 小数点を丸める
   682→  for (const key of Object.keys(ranges)) {
   683→    best[key] = Math.round(best[key] * 100) / 100;
   684→  }
   685→

   688→  console.log('適用するには: debugSetAnswer(' + JSON.stringify(best) + ')');
   689→  return best;
   690→};

   257→

   262→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {

   265→

   267→      if (pixels[(y * W + px) * 4 + 3] > SC.alphaThreshold) {
   268→        topFound = y;
   269→        break;
   270→      }
   271→    }
   272→
   273→    for (let y = H - 1; y >= 0; y--) {
   274→      if (pixels[(y * W + px) * 4 + 3] > SC.alphaThreshold) {
   275→        bottomFound = y;
   276→        break;
   277→      }
   278→    }
   279→
   280→    if (topFound === -1) topFound = idx > 0 ? refTopY[idx - 1] : CAT.backY * H;
   281→    if (bottomFound === -1) bottomFound = idx > 0 ? refBottomY[idx - 1] : CAT.bottomY * H;
   282→
   283→    refTopY[idx] = topFound;
   284→    refBottomY[idx] = bottomFound;
   285→    idx++;
   286→  }
   287→
   288→  smoothContour(refTopY, SC.smoothingRadius);
   289→  smoothContour(refBottomY, SC.smoothingRadius);

   292→function smoothContour(arr, radius) {
   293→  const copy = new Float64Array(arr);
   294→  for (let i = 0; i < arr.length; i++) {
   295→    let sum = 0;
   296→    let count = 0;
   297→    for (let j = -radius; j <= radius; j++) {
   298→      const k = i + j;
   299→      if (k >= 0 && k < arr.length) {
   300→        sum += copy[k];
   301→        count++;
   302→      }
   303→    }
   304→    arr[i] = sum / count;
   305→  }
   306→}

   311→function getEarContribution(xRatio, params, W) {
   312→  const CAT = CFG.cat;
   313→  const ear1X = (CAT.earCenterX - params.earGap / W) * W;
   314→  const ear2X = (CAT.earCenterX + params.earGap / W) * W;
   315→  const x = xRatio * W;
   316→  const ear1 = params.earHeight * Math.exp(-((x - ear1X) ** 2) / (2 * params.earWidth ** 2));
   317→  const ear2 = params.earHeight * Math.exp(-((x - ear2X) ** 2) / (2 * params.earWidth ** 2));
   318→  return ear1 + ear2;
   319→}

   321→function getTailContribution(xRatio, params, W, H) {
   322→  const CAT = CFG.cat;
   323→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   324→  if (t < 0 || t > 1) return 0;
   325→  const tailStart = CFG.topLine.tailStart;
   326→  const tailT = Math.max(0, (t - tailStart) / (1 - tailStart));
   327→  const smoothTail = tailT * tailT * (3 - 2 * tailT);
   328→  return params.tailHeight * smoothTail;

   331→function getFeetContribution(xRatio, params, W, H) {
   332→  const CAT = CFG.cat;
   333→  const BL = CFG.bottomLine;
   334→  const t = (xRatio - CAT.noseX) / (CAT.tailX - CAT.noseX);
   335→  if (t < 0 || t > 1) return 0;
   336→  const feetEnvLeft = Math.min(1, Math.max(0, (t - BL.feetStartT) / BL.feetFadeWidth));
   337→  const feetEnvRight = Math.min(1, Math.max(0, (BL.feetEndT - t) / BL.feetFadeWidth));
   338→  const feetEnv = feetEnvLeft * feetEnvRight;
   339→  const wave = params.feetAmp * Math.sin(params.feetFreq * Math.PI * 2 * t + params.feetPhase);
   340→  return feetEnv * wave;

   343→function interpolateContour(contour, xRatio) {
   344→  if (!contour) return null;
   345→  const W = CFG.canvas.width;
   346→  const startPx = Math.floor(CFG.cat.noseX * W);
   347→  const px = xRatio * W;
   348→  const idx = (px - startPx) / CFG.scoring.sampleStep;
   349→  if (idx < 0 || idx > contour.length - 1) return null;
   350→  const i0 = Math.floor(idx);
   351→  const i1 = Math.min(i0 + 1, contour.length - 1);
   352→  const frac = idx - i0;
   353→  return contour[i0] * (1 - frac) + contour[i1] * frac;

   435→function calculateScore() {
   436→  if (!refTopY || !refBottomY) return 0;
   437→
   438→  const W = CFG.canvas.width;
   439→  const H = CFG.canvas.height;
   440→  const CAT = CFG.cat;
   441→  const SC = CFG.scoring;
   442→  let totalError = 0;
   443→  let count = 0;
   444→
   445→  const startPx = Math.floor(CAT.noseX * W);
   446→  const endPx = Math.ceil(CAT.tailX * W);
   447→
   448→  let idx = 0;
   449→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   450→    const xr = px / W;
   451→    const pTop = getTopY(xr, player, W, H);
   452→    const pBot = getBottomY(xr, player, W, H);
   453→
   454→    const tTop = refTopY[idx];
   455→    const tBot = refBottomY[idx];
   456→    idx++;
   457→
   458→    totalError += (tTop - pTop) ** 2;
   459→    totalError += (tBot - pBot) ** 2;
   460→    count += 2;
   461→  }
   462→
   463→  const rmse = Math.sqrt(totalError / count);
   464→  const score = Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   465→  return score;
   466→}

   572→// 自動フィッティング: answerParamsを最適化
   573→window.autoFit = function () {
   574→  if (!refTopY || !refBottomY) { console.error('extractContours未実行'); return; }
   575→
   576→  const W = CFG.canvas.width;
   577→  const H = CFG.canvas.height;
   578→  const CAT = CFG.cat;
   579→  const SC = CFG.scoring;
   580→  const startPx = Math.floor(CAT.noseX * W);
   581→  const endPx = Math.ceil(CAT.tailX * W);
   582→
   583→  // 現在の数式ベースラインとの誤差を計算
   584→  function calcError(params) {
   585→    let err = 0;
   586→    let count = 0;
   587→    let idx = 0;
   588→    for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   589→      const xr = px / W;
   590→      const t = (xr - CAT.noseX) / (CAT.tailX - CAT.noseX);
   591→      if (t < 0 || t > 1) { idx++; continue; }
   592→
   593→      // 上線: refTopY = baseline - ears - tail を期待
   594→      const ears = getEarContribution(xr, params, W);
   595→      const tail = getTailContribution(xr, params, W, H);
   596→      const feet = getFeetContribution(xr, params, W, H);
   597→
   598→      // 上線の誤差（LUTなしの直接比較）
   599→      // getTopY(数式) = backY + faceY - ears - tail
   600→      // 理想: refTopY[idx] = getTopY(answerParams)
   601→      // ここでは数式ベースラインは使わず、refTopYとの差を最小化
   602→      const backY_val = CAT.backY * H;
   603→      const noseY_val = CAT.noseY * H;
   604→      const faceT = Math.max(0, 1 - t * CFG.topLine.faceDecayRate);
   605→      const faceY = faceT * (noseY_val - backY_val);
   606→      const mathTop = backY_val + faceY - ears - tail;
   607→
   608→      // ※自動フィッティングではLUTを使わず、直接refTopYとparamsでの曲線を比較
   609→      // refTopY = 画像の正解 → paramsでの曲線がこれに近いほど良い
   610→      err += (refTopY[idx] - mathTop) ** 2;
   611→
   612→      // 下線
   613→      const BL = CFG.bottomLine;
   614→      const buttT = (CAT.buttX - CAT.noseX) / (CAT.tailX - CAT.noseX);
   615→      let baseY;
   616→      if (t < BL.faceEndT) {
   617→        baseY = noseY_val + (CAT.bottomY * H - noseY_val) * (t / BL.faceEndT);
   618→      } else {
   619→        baseY = CAT.bottomY * H;
   620→      }
   621→      if (t > BL.buttRiseStart && t <= buttT) {
   622→        const bt = (t - BL.buttRiseStart) / (buttT - BL.buttRiseStart);
   623→        const smooth = bt * bt * (3 - 2 * bt);
   624→        baseY = CAT.bottomY * H + (backY_val - CAT.bottomY * H) * smooth;
   625→      } else if (t > buttT) {
   626→        baseY = backY_val;
   627→      }
   628→      const mathBot = baseY + feet;
   629→      err += (refBottomY[idx] - mathBot) ** 2;
   630→
   631→      count += 2;
   632→      idx++;
   633→    }
   634→    return Math.sqrt(err / count);
   635→  }
   636→
   637→  // グリッドサーチ + 局所最適化
   638→  let best = { ...CFG.answerParams };
   639→  let bestErr = calcError(best);
   640→  console.log('初期誤差:', bestErr.toFixed(2));
   641→
   642→  // 各パラメータの探索範囲
   643→  const ranges = {
   644→    earHeight: [30, 150, 5],
   645→    earGap:    [5, 60, 3],
   646→    earWidth:  [8, 50, 3],
   647→    tailHeight:[20, 150, 5],
   648→    feetAmp:   [10, 70, 3],
   649→    feetFreq:  [2.0, 8.0, 0.2],
   650→    feetPhase: [-3.14, 3.14, 0.15]
   651→  };
   652→
   653→  // 3ラウンドの反復最適化
   654→  for (let round = 0; round < 3; round++) {
   655→    for (const [key, [min, max, step]] of Object.entries(ranges)) {
   656→      let localBest = best[key];
   657→      let localBestErr = bestErr;
   658→      const s = step / (round + 1); // ラウンドごとにステップを細かく
   659→      for (let v = min; v <= max; v += s) {
   660→        const trial = { ...best, [key]: v };
   661→        const e = calcError(trial);
   662→        if (e < localBestErr) {
   663→          localBestErr = e;
   664→          localBest = v;
   665→        }
   666→      }
   667→      best[key] = localBest;
   668→      bestErr = localBestErr;
   669→    }
   670→    console.log(`ラウンド${round + 1} 誤差: ${bestErr.toFixed(2)}`);
   671→  }
   672→
   673→  // 小数点を丸める
   674→  for (const key of Object.keys(ranges)) {
   675→    best[key] = Math.round(best[key] * 100) / 100;
   676→  }
   677→
   678→  console.log('最適化結果 answerParams:', JSON.stringify(best, null, 2));
   679→  console.log('最終誤差:', calcError(best).toFixed(2));
   680→  console.log('適用するには: debugSetAnswer(' + JSON.stringify(best) + ')');
   681→  return best;
   682→};
   683→
   684→init();
   685→
