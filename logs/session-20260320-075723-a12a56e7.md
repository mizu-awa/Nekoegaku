   588→function calculateScore() {
   589→  if (!refTopY || !refBottomY) return 0;
   590→
   591→  const W = CFG.canvas.width;
   592→  const H = CFG.canvas.height;
   593→  const CAT = CFG.cat;
   594→  const SC = CFG.scoring;
   595→  let totalError = 0;
   596→  let count = 0;
   597→
   598→  const startPx = Math.floor(CAT.noseX * W);
   599→  const endPx = Math.ceil(CAT.tailX * W);
   600→
   601→  let idx = 0;
   602→  for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   603→    const xr = px / W;
   604→    const pTop = getTopY(xr, player, W, H);
   605→    const pBot = getBottomY(xr, player, W, H);
   606→
   607→    const tTop = refTopY[idx];
   608→    const tBot = refBottomY[idx];
   609→    idx++;
   610→
   611→    totalError += (tTop - pTop) ** 2;
   612→    totalError += (tBot - pBot) ** 2;
   613→    count += 2;
   614→  }
   615→
   616→  const rmse = Math.sqrt(totalError / count);
   617→  const score = Math.max(0, Math.round(SC.maxScore * Math.max(0, 1 - rmse / SC.maxRmse)));
   618→  return score;
   619→}

   729→// 自動フィッティング: answerParamsを最適化
   730→window.autoFit = function () {
   731→  if (!refTopY || !refBottomY) { console.error('extractContours未実行'); return; }
   732→
   733→  const W = CFG.canvas.width;
   734→  const H = CFG.canvas.height;
   735→  const CAT = CFG.cat;
   736→  const SC = CFG.scoring;
   737→  const startPx = Math.floor(CAT.noseX * W);
   738→  const endPx = Math.ceil(CAT.tailX * W);
   739→
   740→  // 現在の数式ベースラインとの誤差を計算
   741→  function calcError(params) {
   742→    let err = 0;
   743→    let count = 0;
   744→    let idx = 0;
   745→    for (let px = startPx; px <= endPx; px += SC.sampleStep) {
   746→      const xr = px / W;
   747→      const t = (xr - CAT.noseX) / (CAT.tailX - CAT.noseX);
   748→      if (t < 0 || t > 1) { idx++; continue; }
   749→
   750→      // 上線: refTopY = baseline - ears - tail を期待
   751→      const ears = getEarContribution(xr, params, W);
   752→      const tail = getTailContribution(xr, params, W, H);
   753→      const feet = getFeetContribution(xr, params, W, H);
   754→
   755→      // 上線の誤差（LUTなしの直接比較）
   756→      // getTopY(数式) = backY + faceY - ears - tail
   757→      // 理想: refTopY[idx] = getTopY(answerParams)
   758→      // ここでは数式ベースラインは使わず、refTopYとの差を最小化
   759→      const backY_val = CAT.backY * H;
   760→      const noseY_val = CAT.noseY * H;
   761→      const faceT = Math.max(0, 1 - t * CFG.topLine.faceDecayRate);
   762→      const faceY = faceT * (noseY_val - backY_val);
   763→      const mathTop = backY_val + faceY - ears - tail;
   764→
   765→      // ※自動フィッティングではLUTを使わず、直接refTopYとparamsでの曲線を比較
   766→      // refTopY = 画像の正解 → paramsでの曲線がこれに近いほど良い
   767→      err += (refTopY[idx] - mathTop) ** 2;
   768→
   769→      // 下線
   770→      const BL = CFG.bottomLine;
   771→      const buttT = (CAT.buttX - CAT.noseX) / (CAT.tailX - CAT.noseX);
   772→      let baseY;
   773→      if (t < BL.faceEndT) {
   774→        baseY = noseY_val + (CAT.bottomY * H - noseY_val) * (t / BL.faceEndT);
   775→      } else {
   776→        baseY = CAT.bottomY * H;
   777→      }
   778→      if (t > BL.buttRiseStart && t <= buttT) {
   779→        const bt = (t - BL.buttRiseStart) / (buttT - BL.buttRiseStart);
   780→        const smooth = bt * bt * (3 - 2 * bt);
   781→        baseY = CAT.bottomY * H + (backY_val - CAT.bottomY * H) * smooth;
   782→      } else if (t > buttT) {
   783→        const tailProgress = (t - buttT) / (1 - buttT);
   784→        const smooth2 = tailProgress * tailProgress * (3 - 2 * tailProgress);
   785→        const tailTipY = backY_val - getTailContribution(CAT.tailX, trial, W, H);
   786→        baseY = backY_val + (tailTipY - backY_val) * smooth2;
   787→      }
   788→      const mathBot = baseY + feet;
   789→      err += (refBottomY[idx] - mathBot) ** 2;
   790→
   791→      count += 2;
   792→      idx++;
   793→    }
   794→    return Math.sqrt(err / count);
   795→  }
   796→
   797→  // グリッドサーチ + 局所最適化
   798→  let best = { ...CFG.answerParams };
   799→  let bestErr = calcError(best);
   800→  console.log('初期誤差:', bestErr.toFixed(2));
   801→
   802→  // 各パラメータの探索範囲
   803→  const ranges = {
   804→    earHeight: [30, 150, 5],
   805→    earGap:    [5, 60, 3],
   806→    earWidth:  [8, 50, 3],
   807→    tailHeight:[20, 150, 5],
   808→    feetAmp:   [10, 70, 3],
   809→    feetFreq:  [2.0, 8.0, 0.2],
   810→    feetPhase: [-3.14, 3.14, 0.15]
   811→  };
   812→
   813→  // 3ラウンドの反復最適化
   814→  for (let round = 0; round < 3; round++) {
   815→    for (const [key, [min, max, step]] of Object.entries(ranges)) {
   816→      let localBest = best[key];
   817→      let localBestErr = bestErr;
   818→      const s = step / (round + 1); // ラウンドごとにステップを細かく
   819→      for (let v = min; v <= max; v += s) {
   820→        const trial = { ...best, [key]: v };
   821→        const e = calcError(trial);
   822→        if (e < localBestErr) {
   823→          localBestErr = e;
   824→          localBest = v;
   825→        }
   826→      }
   827→      best[key] = localBest;
   828→      bestErr = localBestErr;
   829→    }
   830→    console.log(`ラウンド${round + 1} 誤差: ${bestErr.toFixed(2)}`);
   831→  }
   832→
   833→  // 小数点を丸める
   834→  for (const key of Object.keys(ranges)) {
   835→    best[key] = Math.round(best[key] * 100) / 100;
   836→  }
   837→
   838→  console.log('最適化結果 answerParams:', JSON.stringify(best, null, 2));
   839→  console.log('最終誤差:', calcError(best).toFixed(2));
   840→  console.log('適用するには: debugSetAnswer(' + JSON.stringify(best) + ')');
   841→  return best;
   842→};
